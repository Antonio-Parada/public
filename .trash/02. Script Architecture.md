#!/bin/bash
# Force path to ensure scripts are found when launched via Hyprland
export PATH="$HOME/.local/bin:$PATH"

OPTS="  Samsung Flip\n  Pixel 8 Pro"

# Launch Wofi Menu
SELECTED=$(echo -e "$OPTS" | wofi --dmenu --cache-file /dev/null --prompt "Select Device" --width 300 --height 200)

case $SELECTED in
  "  Samsung Flip")
    exec flip-ctl ;;
  "  Pixel 8 Pro")
    exec pixel-ctl ;;
esac
